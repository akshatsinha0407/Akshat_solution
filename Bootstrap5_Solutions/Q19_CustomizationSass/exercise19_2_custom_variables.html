Sample _variables.scss Override File
// Custom Bootstrap Variables — scss/_variables.scss
$primary:          #e74c3c;
$secondary:        #95a5a6;
$success:          #27ae60;
$info:             #2980b9;
$warning:          #f39c12;
$danger:           #c0392b;

$border-radius:    1rem;
$border-radius-lg: 1.5rem;
$border-radius-sm: 0.75rem;

$font-size-base:    1.05rem;
$headings-font-weight: 700;

$enable-gradients:  true;
$enable-shadows:    true;
$enable-negative-margins: true;

// Then import Bootstrap AFTER overrides:
@import "../node_modules/bootstrap/scss/bootstrap";
